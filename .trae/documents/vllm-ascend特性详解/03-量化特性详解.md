# 量化特性详解

## 1. 特性概述

### 1.1 什么是量化？

量化是将模型权重和激活值从高精度（如FP32/FP16）转换为低精度（如INT8/INT4）的技术，在保持模型精度的同时，大幅降低显存占用和计算开销。

### 1.2 vLLM-Ascend支持的量化方法

| 量化方法 | 权重精度 | 激活精度 | 显存降低 | 性能提升 | 精度损失 |
|---------|---------|---------|---------|---------|---------|
| W4A16-AWQ | INT4 | FP16/BF16 | 75% | 80% (小并发) | <2% |
| W8A8-SmoothQuant | INT8 | INT8 | 50% | 30% | <1.5% |
| W8A16-GPTQ | INT8 | FP16/BF16 | 50% | 20% | <1% |
| Kv8 | INT8 | - | 50% | 15-25% | 极小 |

## 2. W4A16-AWQ量化

### 2.1 设计方案

#### AWQ原理
```
AWQ (Activation-aware Weight Quantization)

核心思想：
1. 权重不是同等重要的，部分权重对激活值影响更大
2. 保护重要权重，减少量化误差
3. 基于激活值分布确定权重重要性

公式：
W_quant = round(W / scale) * scale
scale = max(|W|) / (2^(bits-1) - 1)

关键优化：
- 基于激活值分布确定scale
- 保护重要权重通道
- 使用per-channel量化
```

#### 整体架构
```
┌─────────────────────────────────────────────────────────┐
│                    AWQ量化流程                           │

│                                                         │
```

```python
import torch
import torch.nn as nn
from typing import List, Tuple

class AWQQuantizer:
    """
    AWQ量化器
    
    核心步骤：
    1. 收集激活值分布
    2. 计算权重重要性
    3. 确定量化参数
    4. 量化权重
    """
    
    def __init__(self, 
                 model: nn.Module,
                 w_bit: int = 4,
                 q_group_size: int = 128):
        self.model = model
        self.w_bit = w_bit
        self.q_group_size = q_group_size
        
        # 量化参数
        self.scales = {}
        self.zeros = {}
        
    def calibrate(self, calib_data: List[torch.Tensor]):
        """
        校准：收集激活值分布
        
        Args:
            calib_data: 校准数据集
        """
        # 注册hook收集激活值
        activations = {}
        
        def hook_fn(name):
            def hook(module, input, output):
                activations[name] = input[0].detach()
            return hook
        
        # 注册hook
        hooks = []
        for name, module in self.model.named_modules():
            if isinstance(module, nn.Linear):
                hooks.append(module.register_forward_hook(hook_fn(name)))
        
        # 推理收集激活值
        with torch.no_grad():
            for data in calib_data:
                self.model(data)
        
        # 移除hook
        for hook in hooks:
            hook.remove()
        
        # 计算量化参数
        self._compute_quant_params(activations)
    
    def _compute_quant_params(self, activations: dict):
        """
        计算量化参数
        
        关键：基于激活值分布确定权重重要性
        """
        for name, module in self.model.named_modules():
            if isinstance(module, nn.Linear):
                weight = module.weight.data
                activation = activations.get(name)
                
                if activation is not None:
                    # 计算权重重要性
                    # 基于激活值的幅度
                    importance = activation.abs().mean(dim=0)  # [out_features]
                    
                    # 计算scale
                    # per-channel量化
                    weight_max = weight.abs().max(dim=1)[0]  # [out_features]
                    scale = weight_max / (2 ** (self.w_bit - 1) - 1)
                    
                    # 保护重要通道
                    # 对重要通道使用更小的scale
                    importance_norm = importance / importance.max()
                    scale = scale * (1 + importance_norm * 0.1)
                    
                    self.scales[name] = scale
                    self.zeros[name] = torch.zeros_like(scale)
    
    def quantize(self):
        """
        量化权重
        
        INT4量化公式：
        W_int4 = round(W_fp16 / scale)
        W_quant = W_int4 * scale
        """
        quantized_weights = {}
        
        for name, module in self.model.named_modules():
            if isinstance(module, nn.Linear):
                weight = module.weight.data
                scale = self.scales[name]
                
                # 量化
                weight_int = torch.round(weight / scale.unsqueeze(1))
                
                # Clamping
                weight_int = torch.clamp(
                    weight_int,
                    -2 ** (self.w_bit - 1),
                    2 ** (self.w_bit - 1) - 1
                )
                
                # 保存量化权重
                quantized_weights[name] = {
                    'weight': weight_int.to(torch.int8),
                    'scale': scale,
                    'zero': self.zeros[name]
                }
        
        return quantized_weights
    
    def save_quantized_model(self, save_path: str):
        """保存量化模型"""
        quantized_weights = self.quantize()
        
        # 保存
        torch.save({
            'quantized_weights': quantized_weights,
            'scales': self.scales,
            'zeros': self.zeros,
            'w_bit': self.w_bit,
            'q_group_size': self.q_group_size,
        }, save_path)
```

#### 在线推理
```python
class AWQLinear(nn.Module):
    """
    AWQ量化线性层
    
    在线推理：
    1. 加载INT4权重
    2. 输入时反量化
    3. 计算输出
    """
    
    def __init__(self, 
                 in_features: int,
                 out_features: int,
                 w_bit: int = 4):
        super().__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.w_bit = w_bit
        
        # 量化权重（INT4存储为INT8）
        self.weight = nn.Parameter(
            torch.zeros(out_features, in_features, dtype=torch.int8),
            requires_grad=False
        )
        
        # 量化参数
        self.scale = nn.Parameter(
            torch.ones(out_features, dtype=torch.float16),
            requires_grad=False
        )
        
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        前向传播
        
        流程：
        1. 反量化权重（INT4 -> FP16）
        2. 计算 x @ W^T
        """
        # 反量化
        weight_fp16 = self.weight.float() * self.scale.unsqueeze(1)
        
        # 计算
        output = torch.nn.functional.linear(x, weight_fp16)
        
        return output


class AWQModelForCausalLM(nn.Module):
    """
    AWQ量化模型
    """
    
    def __init__(self, model_path: str):
        super().__init__()
        
        # 加载量化权重
        checkpoint = torch.load(model_path)
        quantized_weights = checkpoint['quantized_weights']
        
        # 构建模型
        self.layers = nn.ModuleList()
        
        for name, params in quantized_weights.items():
            if 'weight' in name:
                # 创建AWQLinear层
                layer = AWQLinear(
                    in_features=params['weight'].shape[1],
                    out_features=params['weight'].shape[0],
                    w_bit=checkpoint['w_bit']
                )
                
                # 加载权重
                layer.weight.data = params['weight']
                layer.scale.data = params['scale']
                
                self.layers.append(layer)
    
    def forward(self, input_ids: torch.Tensor) -> torch.Tensor:
        """前向传播"""
        # Embedding
        hidden_states = self.embed(input_ids)
        
        # Transformer layers
        for layer in self.layers:
            hidden_states = layer(hidden_states)
        
        # LM head
        logits = self.lm_head(hidden_states)
        
        return logits
```

```python
import torch_npu

class NPUAWQLinear(nn.Module):
    """
    针对Ascend NPU优化的AWQ线性层

    1. 使用NPU量化算子
    2. 融合反量化和计算
    3. 减少内存访问

    def __init__(self, in_features, out_features, w_bit=4):

        self.w_bit = w_bit

        )
        
        return output
```

## 3. W8A8-SmoothQuant量化

### 3.1 设计方案

#### SmoothQuant原理
```
SmoothQuant: 平滑激活值和权重的量化难度

核心思想：
1. 将激活值的量化难度迁移到权重
2. 平衡激活值和权重的量化误差
3. 使用平滑因子调整分布

公式：
Y = X @ W
  = (X / s) @ (s * W)  # 数学等价
  = X_smooth @ W_smooth

其中：
- s: 平滑因子
- X_smooth = X / s  # 激活值平滑
- W_smooth = s * W  # 权重平滑

优势：
- 激活值分布更平滑，易于量化
- 权重分布更集中，量化误差小
- 整体量化精度更高
```

### 3.2 关键代码解读

```python
class SmoothQuantizer:
    """
    SmoothQuant量化器
    """
    
    def __init__(self, model, alpha=0.5):
        """
        Args:
            model: 原始模型
            alpha: 平滑因子系数 (0-1)
                   alpha=0: 仅量化权重
                   alpha=1: 仅量化激活
                   alpha=0.5: 平衡（推荐）
        """
        self.model = model
        self.alpha = alpha
        
    def smooth_model(self, calib_data):
        """
        平滑模型
        
        步骤：
        1. 收集激活值和权重分布
        2. 计算平滑因子
        3. 调整权重和激活值
        """
        # 收集激活值
        activations = self._collect_activations(calib_data)
        
        # 对每个线性层平滑
        for name, module in self.model.named_modules():
            if isinstance(module, nn.Linear):
                # 获取激活值
                activation = activations[name]
                
                # 计算平滑因子
                # s = max(|X|)^alpha / max(|W|)^(1-alpha)
                act_max = activation.abs().max(dim=0)[0]  # [in_features]
                weight_max = module.weight.abs().max(dim=0)[0]  # [in_features]
                
                smooth_factor = (
                    act_max.pow(self.alpha) / 
                    weight_max.pow(1 - self.alpha)
                )
                
                # 平滑权重
                module.weight.data = module.weight.data * smooth_factor.unsqueeze(0)
                
                # 保存平滑因子（用于推理时平滑激活值）
                module.smooth_factor = smooth_factor
    
    def quantize(self):
        """
        量化平滑后的模型
        
        W8A8量化：
        - 权重: FP16 -> INT8
        - 激活: FP16 -> INT8 (动态量化)
        """
        for name, module in self.model.named_modules():
            if isinstance(module, nn.Linear):
                # 权重量化
                weight = module.weight.data
                weight_max = weight.abs().max()
                weight_scale = weight_max / 127.0
                
                weight_int8 = torch.round(weight / weight_scale).to(torch.int8)
                
                # 保存量化参数
                module.weight_quant = weight_int8
                module.weight_scale = weight_scale


class SmoothQuantLinear(nn.Module):
    """
    SmoothQuant量化线性层
    
    W8A8量化：
    - 权重: INT8 (静态量化)
    - 激活: INT8 (动态量化)
    """
    
    def __init__(self, in_features, out_features):
        super().__init__()
        self.in_features = in_features
        self.out_features = out_features
        
        # INT8权重
        self.weight = nn.Parameter(
            torch.zeros(out_features, in_features, dtype=torch.int8),
            requires_grad=False
        )
        
        # 权重scale
        self.weight_scale = nn.Parameter(
            torch.tensor(1.0, dtype=torch.float32),
            requires_grad=False
        )
        
        # 平滑因子
        self.smooth_factor = nn.Parameter(
            torch.ones(in_features, dtype=torch.float16),
            requires_grad=False
        )
        
    def forward(self, x):
        """
        前向传播
        
        流程：
        1. 平滑激活值
        2. 动态量化激活值
        3. INT8矩阵乘法
        4. 反量化输出
        """
        # 1. 平滑激活值
        x_smooth = x / self.smooth_factor
        
        # 2. 动态量化激活值
        x_max = x_smooth.abs().max()
        x_scale = x_max / 127.0
        x_int8 = torch.round(x_smooth / x_scale).to(torch.int8)
        
        # 3. INT8矩阵乘法
        # output_int32 = x_int8 @ weight_int8
        output_int32 = torch.matmul(
            x_int8.float(),
            self.weight.float()
        )
        
        # 4. 反量化
        output_scale = x_scale * self.weight_scale
        output = output_int32 * output_scale
        
        return output.to(torch.float16)


class NPUSmoothQuantLinear(nn.Module):
    """
    针对Ascend NPU优化的SmoothQuant线性层
    """
    
    def forward(self, x):
        """
        NPU优化的W8A8计算
        
        使用torch_npu的INT8矩阵乘法算子
        """
        # 平滑激活值
        x_smooth = x / self.smooth_factor
        
        # 使用NPU INT8算子
        # npu_dynamic_quant_matmul: 动态量化矩阵乘法
        output = torch_npu.npu_dynamic_quant_matmul(
            x_smooth,              # FP16激活
            self.weight,           # INT8权重
            self.weight_scale,     # 权重scale
            output_dtype=torch.float16
        )
        
        return output
```

## 4. Kv8量化

### 4.1 设计方案

```
KV Cache量化

目标：
- 降低KV Cache显存占用
- 支持更长序列
- 减少内存带宽压力

方法：
- 静态量化：离线确定scale
- 动态量化：在线确定scale
- Per-token量化：每个token独立scale
```

### 4.2 关键代码解读

```python
class KVCacheQuantizer:
    """
    KV Cache量化器
    """
    
    def __init__(self, kv_bit=8):
        self.kv_bit = kv_bit
        
    def quantize_kv_cache(self, key, value):
        """
        量化KV Cache
        
        Args:
            key: [batch, num_heads, seq_len, head_dim], FP16
            value: [batch, num_heads, seq_len, head_dim], FP16
        
        Returns:
            key_int8: INT8量化后的key
            value_int8: INT8量化后的value
            key_scale: key的scale
            value_scale: value的scale
        """
        # Per-token量化
        # 对每个token计算scale
        
        # Key量化
        key_max = key.abs().max(dim=-1, keepdim=True)[0]  # [batch, num_heads, seq_len, 1]
        key_scale = key_max / 127.0
        key_int8 = torch.round(key / key_scale).to(torch.int8)
        
        # Value量化
        value_max = value.abs().max(dim=-1, keepdim=True)[0]
        value_scale = value_max / 127.0
        value_int8 = torch.round(value / value_scale).to(torch.int8)
        
        return key_int8, value_int8, key_scale, value_scale
    
    def dequantize_kv_cache(self, key_int8, value_int8, key_scale, value_scale):
        """
        反量化KV Cache
        """
        key_fp16 = key_int8.float() * key_scale
        value_fp16 = value_int8.float() * value_scale
        
        return key_fp16.to(torch.float16), value_fp16.to(torch.float16)


class QuantizedPagedAttention:
    """
    支持KV8量化的PagedAttention
    """
    
    def __init__(self, num_heads, head_dim, block_size, kv_bit=8):
        self.num_heads = num_heads
        self.head_dim = head_dim
        self.block_size = block_size
        self.kv_bit = kv_bit
        
        # Block存储（INT8）
        self.key_cache = torch.zeros(
            num_blocks, block_size, num_heads, head_dim,
            dtype=torch.int8,
            device='npu'
        )
        
        self.value_cache = torch.zeros(
            num_blocks, block_size, num_heads, head_dim,
            dtype=torch.int8,
            device='npu'
        )
        
        # Scale存储（FP16）
        self.key_scale_cache = torch.zeros(
            num_blocks, block_size, num_heads, 1,
            dtype=torch.float16,
            device='npu'
        )
        
        self.value_scale_cache = torch.zeros(
            num_blocks, block_size, num_heads, 1,
            dtype=torch.float16,
            device='npu'
        )
    
    def forward(self, query, block_tables, context_lens):
        """
        量化KV Cache的Attention计算
        
        流程：
        1. 收集KV Cache（INT8）
        2. 反量化
        3. Attention计算
        """
        # 收集KV Cache
        key_cache = self.key_cache[block_tables]  # [batch, num_blocks, block_size, num_heads, head_dim]
        value_cache = self.value_cache[block_tables]
        
        key_scale = self.key_scale_cache[block_tables]
        value_scale = self.value_scale_cache[block_tables]
        
        # 反量化
        key_fp16 = key_cache.float() * key_scale
        value_fp16 = value_cache.float() * value_scale
        
        # Attention计算
        # 使用NPU融合算子
        output = torch_npu.npu_fusion_attention(
            query, key_fp16, value_fp16,
            head_num=self.num_heads,
            input_layout="BNSD",
            scale=1.0 / math.sqrt(self.head_dim)
        )[0]
        
        return output


class NPUQuantizedKVCache:
    """
    针对Ascend NPU优化的量化KV Cache
    
    优化点：
    1. 使用NPU量化算子
    2. 融合反量化和Attention
    3. 减少内存访问
    """
    
    def forward(self, query, block_tables):
        """
        NPU优化的量化KV Cache Attention
        
        使用torch_npu的融合算子
        """
        # 使用NPU融合算子
        # npu_quant_fusion_attention: 量化KV Cache的融合Attention

            query,                           # FP16 query
            self.key_cache,                  # INT8 key cache
            self.value_cache,                # INT8 value cache
            self.key_scale_cache,            # key scale
            self.value_scale_cache,          # value scale
            block_tables,                    # block table
            head_num=self.num_heads,
            input_layout="BNSD",
            scale=1.0 / math.sqrt(self.head_dim)
        )
        
        return output
```

```python
def compare_memory_usage():
    """
    不同量化方法的显存占用对比
    
    以Qwen-72B为例：
    - 原始FP16: 144GB

    # W4A16
    w4a16_memory = model_size * 0.5  # 0.5 bytes per parameter
    
    # W8A8 + Kv8
    # 权重: 50%降低
    # KV Cache: 50%降低（假设KV Cache占30%）
    w8a8_kv8_memory = model_size * 1 * 0.7 + model_size * 2 * 0.3 * 0.5

    print(f"W4A16: {w4a16_memory:.2f} GB ({(1-w4a16_memory/fp16_memory)*100:.1f}% 降低)")
    print(f"W8A8+Kv8: {w8a8_kv8_memory:.2f} GB ({(1-w8a8_kv8_memory/fp16_memory)*100:.1f}% 降低)")
```

### 6.1 量化方法选择

```python
def select_quantization_method(requirements):
    """
    根据需求选择量化方法
    
    Args:
        requirements: dict包含
            - accuracy_sensitive: 是否对精度敏感
            - memory_constrained: 是否显存受限
            - latency_sensitive: 是否对延迟敏感
            - throughput_oriented: 是否追求吞吐量
    """
    if requirements['accuracy_sensitive']:
        # 精度敏感：选择W8A16-GPTQ
        return 'W8A16-GPTQ'
    
    elif requirements['memory_constrained']:
        # 显存受限：选择W4A16-AWQ
        return 'W4A16-AWQ'
    
    elif requirements['latency_sensitive']:
        # 延迟敏感：选择W4A16-AWQ
        return 'W4A16-AWQ'
    
    elif requirements['throughput_oriented']:
        # 追求吞吐量：选择W8A8-SmoothQuant + Kv8
        return 'W8A8-SmoothQuant+Kv8'
    
    else:
        # 平衡场景：选择W8A8-SmoothQuant
        return 'W8A8-SmoothQuant'
```

## 7. 总结

量化是降低大模型部署成本的关键技术，vLLM-Ascend支持多种量化方法：

1. **W4A16-AWQ**：极致压缩，适合显存受限场景
2. **W8A8-SmoothQuant**：平衡精度和性能，适合大多数场景
3. **W8A16-GPTQ**：精度优先，适合精度敏感场景
4. **Kv8**：KV Cache量化，支持更长序列

在Ascend NPU上，通过使用torch_npu的量化算子，进一步提升了量化推理的性能。
